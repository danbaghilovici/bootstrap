HTML Structure Reference

Copy this HTML structure to use the dropdown component.

<!-- Basic dropdown -->
<div class="dropdown">
  <button class="btn btn-secondary dropdown-toggle" type="button"
          data-bs-toggle="dropdown" aria-expanded="false">
    Dropdown button
  </button>
  <ul class="dropdown-menu">
    <li><a class="dropdown-item" href="#">Action</a></li>
    <li><a class="dropdown-item" href="#">Another action</a></li>
  </ul>
</div>

<!-- With header and divider -->
<ul class="dropdown-menu">
  <li><h6 class="dropdown-header">Section Header</h6></li>
  <li><a class="dropdown-item" href="#">Action</a></li>
  <li><hr class="dropdown-divider"></li>
  <li><a class="dropdown-item" href="#">Another action</a></li>
</ul>

<!-- Item states -->
<a class="dropdown-item active" href="#">Active item</a>
<a class="dropdown-item disabled" aria-disabled="true">Disabled item</a>

<!-- Dark variant -->
<ul class="dropdown-menu dropdown-menu-dark">...</ul>

<!-- Directions -->
<div class="btn-group dropup">...</div>
<div class="btn-group dropend">...</div>
<div class="btn-group dropstart">...</div>

<!-- Split button -->
<div class="btn-group">
  <button type="button" class="btn btn-primary">Action</button>
  <button type="button" class="btn btn-primary dropdown-toggle dropdown-toggle-split"
          data-bs-toggle="dropdown" aria-expanded="false">
    <span class="visually-hidden">Toggle Dropdown</span>
  </button>
  <ul class="dropdown-menu">...</ul>
</div>

CSS Classes Reference

Class Description
.dropdown Container for dropdown trigger and menu
.dropdown-toggle Trigger button/link with caret icon
.dropdown-menu Menu container
.dropdown-item Menu item (link or button)
.dropdown-header Section header text
.dropdown-divider Horizontal separator line
.dropdown-menu-dark Dark theme variant
.active Active/selected item state
.disabled Disabled item state
.dropup Opens menu upward
.dropend Opens menu to the right
.dropstart Opens menu to the left
.dropdown-toggle-split Split button dropdown toggle
