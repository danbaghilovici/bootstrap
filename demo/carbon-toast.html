HTML Structure Reference

Copy this HTML structure to use the toast component.

<!-- Toast container (position fixed) -->
<div class="toast-container position-fixed top-0 end-0 p-3">

  <!-- Default toast -->
  <div class="toast" role="alert" aria-live="assertive" aria-atomic="true">
    <div class="toast-header">
      <strong class="me-auto">Notification</strong>
      <small class="text-body-secondary">just now</small>
      <button type="button" class="btn-close" data-bs-dismiss="toast" aria-label="Close"></button>
    </div>
    <div class="toast-body">
      Your message here.
    </div>
  </div>

</div>

<!-- Toast variants (add class to .toast element) -->
<div class="toast toast-success">...</div>  <!-- Green border -->
<div class="toast toast-warning">...</div>  <!-- Yellow border -->
<div class="toast toast-danger">...</div>   <!-- Red border -->
<div class="toast toast-info">...</div>     <!-- Blue-70 border -->

<!-- Initialize toast via JavaScript -->
<script>
  const toastEl = document.getElementById('myToast');
  const toast = new bootstrap.Toast(toastEl);
  toast.show();
</script>

<!-- Or use data attributes -->
<button type="button" data-bs-toggle="toast" data-bs-target="#myToast">
  Show Toast
</button>
