HTML Structure Reference

Copy these HTML structures to use form components.

<!-- Basic input with label and helper text -->
<label for="inputId" class="form-label">Label Text</label>
<input type="text" class="form-control" id="inputId" placeholder="Placeholder">
<div class="form-text">Helper text goes here.</div>

<!-- Size variants -->
<input type="text" class="form-control form-control-sm">  <!-- 32px -->
<input type="text" class="form-control">                  <!-- 40px -->
<input type="text" class="form-control form-control-lg">  <!-- 48px -->

<!-- Select dropdown -->
<select class="form-select">
  <option selected>Choose...</option>
  <option value="1">Option 1</option>
</select>

<!-- Input group -->
<div class="input-group">
  <span class="input-group-text">@</span>
  <input type="text" class="form-control" placeholder="Username">
</div>

<!-- Validation states -->
<input type="text" class="form-control is-valid">
<div class="valid-feedback">Success message</div>

<input type="text" class="form-control is-invalid">
<div class="invalid-feedback">Error message</div>

<!-- Disabled state -->
<input type="text" class="form-control" disabled>

CSS Classes Reference

Class Description
.form-control Base class for text inputs and textareas
.form-control-sm Small input (32px height)
.form-control-lg Large input (48px height)
.form-select Base class for select dropdowns
.form-select-sm Small select (32px height)
.form-select-lg Large select (48px height)
.form-label Label styling (12px, gray)
.form-text Helper text styling (12px, gray)
.input-group Container for input with addons
.input-group-text Text addon for input groups
.is-valid Valid state (green border)
.is-invalid Invalid state (red border)
.valid-feedback Success message text (green)
.invalid-feedback Error message text (red)
