HTML Structure Reference

Copy this HTML structure to use the pagination nav component.

<nav class="pagination-nav" aria-label="Table pagination">
  <div class="pagination-nav-left">
    <label class="pagination-nav-label" for="pageSize">Items per page:</label>
    <select class="pagination-nav-select" id="pageSize">
      <option value="10">10</option>
      <option value="25">25</option>
      <option value="50">50</option>
    </select>
    <span class="pagination-nav-text">1-10 of 100 items</span>
  </div>
  <div class="pagination-nav-right">
    <select class="pagination-nav-select pagination-nav-select-page"
            id="pageNum" aria-label="Page number">
      <option value="1">1</option>
      <!-- ... more pages ... -->
    </select>
    <span class="pagination-nav-text">of 10 pages</span>
    <button class="pagination-nav-btn" disabled aria-label="Previous page">
      <svg viewBox="0 0 16 16">
        <path d="M10 12L6 8l4-4" stroke="currentColor"
              stroke-width="1.5" fill="none"/>
      </svg>
    </button>
    <button class="pagination-nav-btn" aria-label="Next page">
      <svg viewBox="0 0 16 16">
        <path d="M6 12l4-4-4-4" stroke="currentColor"
              stroke-width="1.5" fill="none"/>
      </svg>
    </button>
  </div>
</nav>

CSS Classes Reference

Class Description
.pagination-nav Main container (uses flexbox, default 40px height)
.pagination-nav-sm Small variant modifier (32px height)
.pagination-nav-lg Large variant modifier (48px height)
.pagination-nav-left Left section container (items per page)
.pagination-nav-right Right section container (page navigation)
.pagination-nav-label Text label (e.g., "Items per page:")
.pagination-nav-select Styled select dropdown
.pagination-nav-select-page Compact page number select
.pagination-nav-text Text spans (item ranges, page counts)
.pagination-nav-btn Prev/Next navigation buttons
